class Solution {
    public int numJewelsInStones(String jewels, String stones) {
        HashSet<Character> jewel = new HashSet<>();
        char[] jw = jewels.toCharArray();
        char[] st = stones.toCharArray();
        for(char c : jw) jewel.add(c);
        int ans =0;
        if(stones.length()==0) return ans;
        for(char c : st){
            if(jewel.contains(c)) {
                  ans++;
            }
         }
        return ans;
    }
}
